18/*******************************************************************************
19 * TRC_CFG_NTASK, TRC_CFG_NISR, TRC_CFG_NQUEUE, TRC_CFG_NSEMAPHORE...
20 *
21 * A group of macros which should be defined as integer values, zero or larger.
22 *
23 * These define the capacity of the Object Property Table, i.e., the maximum
24 * number of objects active at any given point, within each object class (e.g.,
25 * task, queue, semaphore, ...).
26 *
27 * If tasks or other objects are deleted in your system, this
28 * setting does not limit the total amount of objects created, only the number
29 * of objects that have been successfully created but not yet deleted.
30 *
31 * Using too small values will cause vTraceError to be called, which stores an
32 * error message in the trace that is shown when opening the trace file. The
33 * error message can also be retrieved using xTraceGetLastError.
34 *
35 * It can be wise to start with large values for these constants,
36 * unless you are very confident on these numbers. Then do a recording and
37 * check the actual usage by selecting View menu -> Trace Details ->
38 * Resource Usage -> Object Table.
39 ******************************************************************************/
40#define TRC_CFG_NTASK 15
41#define TRC_CFG_NISR 5
42#define TRC_CFG_NQUEUE 10
43#define TRC_CFG_NSEMAPHORE 10
44#define TRC_CFG_NMUTEX 10
45#define TRC_CFG_NFLAG 5
46#define TRC_CFG_NMEM 5
47#define TRC_CFG_NTASK_SEM TRC_CFG_NTASK
48#define TRC_CFG_NTASK_Q TRC_CFG_NTASK
49
50/******************************************************************************
51 * TRC_CFG_NAME_LEN_TASK, TRC_CFG_NAME_LEN_QUEUE, ...
52 *
53 * Macros that specify the maximum lengths (number of characters) for names of
54 * kernel objects, such as tasks and queues. If longer names are used, they will
55 * be truncated when stored in the recorder.
56 *****************************************************************************/
57#define TRC_CFG_NAME_LEN_TASK 15
58#define TRC_CFG_NAME_LEN_ISR 15
59#define TRC_CFG_NAME_LEN_QUEUE 15
60#define TRC_CFG_NAME_LEN_SEMAPHORE 15
61#define TRC_CFG_NAME_LEN_MUTEX 15
62#define TRC_CFG_NAME_LEN_FLAG 32
63#define TRC_CFG_NAME_LEN_MEM 32
64#define TRC_CFG_NAME_LEN_TASK_SEM 32
65#define TRC_CFG_NAME_LEN_TASK_Q 32
